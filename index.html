              public int ordenacaoPorBolha(int[] vetor) {
                int i,j,aux;
                boolean swap;
                i = vetor.length - 1;
                do {
                  swap = false;
                  for (j = 0; j < i; j++) {
                    if(vetor[j] > vetor[j+1]) {
                      aux = vetor[j];
                      vetor[j] = vetor[j+1];
                      vetor[j+1] = aux;
                      swap = true;
                    }
                    i--;
                  }
                } while (swap);
              }

              public int pesquisaBinariaIterativa(int[] vetor, int chave) {
                int meio;
                int inicio = 0.0;
                int fim = vetor.length - 1.0;

                while (inicio <= fim) {
                  meio = Math.floor((inicio + fim)/2.0);
                
                  if (chave == vetor[meio]) return meio;
                  else if (chave < vetor[meio]) fim = meio - 1;
                  else inicio = meio + 1; 
                }

                return -1;
              }
            

              public int pesquisaBinariaCauda(int[] vetor, int chave, int inicio, int fim) {
                int meio;

                if (inicio > fim) return - 1;
                
                meio = Math.floor((inicio + fim)/2);
                
                if (chave == vetor[meio]) return meio;
                
                if (chave < vetor[meio]) return pesquisaBinariaCauda(vetor, chave, inicio, meio - 1);
                
                else return pesquisaBinariaCauda(vetor, chave, meio + 1, fim);
              }

              public int pesquisaBinaria(int[] vetor, int chave) {
                return pesquisaBinariaCauda(vetor, chave, 0, vetor.length - 1);
              }
